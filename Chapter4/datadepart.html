一、数据读取

​ 1.读取csv文件（csv文件就是用逗号分隔的文本文件；使用python的csv模块来处理csv文件）

import pytest
import csv

#定义获取数据函数
def get_data():
    with open("test.csv") as f:
        data=csv.reader(f)
        #定义列表存储数据
        my_data=[]
        for row in data:
            my_data.extend(row)#extend方法添加数据
        return my_data

@pytest.mark.parametrize('name',get_data())
def test_01(name):
    print(name)#获取csv数据并打印


if __name__ == '__main__':
    pytest.main(['-s','test_csv.py'])

​ 3.读取execl文件（需要通过安装xlrd模块来处理数据）

​ 如果运行代码报错 （raise XLRDError(FILE_FORMAT_DESCRIPTIONS[file_format]+'; not supported') 安装pyexcel-xls即可解决

import pytest
import xlrd

#定义获取数据函数
def get_data():
    #要打开的文件名
    filename="data.xlsx"
    #通过xlrd 中open_workbook加载文件数据
    wb=xlrd.open_workbook(filename)
    #定义一个变量获取excel第一sheet数据
    sheet=wb.sheet_by_index(0)
    #分别获取横纵数据
    rows=sheet.nrows
    cols=sheet.ncols
    #定义一个列表存储数据
    lst=[]
    #遍历数据并进行存储
    for row in range(rows):
        for col in range(cols):
            cell_data=sheet.cell_value(row,col)
            lst.append(cell_data)
    return lst
@pytest.mark.parametrize('name',get_data())
def test_001(name):
    print(name)

if __name__ == '__main__':
    pytest.main(['-sv', 'test_excel.py'])

​ 4.读取数据库数据（利用mysqlclient模块）

import pytest
import MySQLdb  # 安装了mysqlclient第三方库即可

# 配置连接数据库信息
conn = MySQLdb.connect(
    user='xxx',
    passwd='xxxxxxxxx',
    host="localhost",
    port=3306,
    db='test_db'

)


# 定义获取数据函数

def get_data():
    # sql语句
    query_sql = "select id,name,pwd from User_info"
    # 定义列表存储
    lst = []
    try:
        # 连接数据库
        cursor = conn.cursor()
        # 执行sql语句
        cursor.execute(query_sql)
        # 返回多个行记录（元组）
        r = cursor.fetchall()
        # 遍历数据 存储到lst中
        for x in r:
            u = (x[0], x[1], x[2])  # 将其三个值以元组返回
            lst.append(u)
        return lst
    finally:
        # 游标关闭
        cursor.close()
        # 连接关闭
        conn.close()


@pytest.mark.parametrize("id,name,pwd", get_data())
def test_001(id, name, pwd):
    print(id, name, pwd)


if __name__ == '__main__':
    pytest.main(['-s', 'test_DB.py'])

二、DDT思想

1.安装ddt模块

pip install ddt

2.DDT中使用@ddt、@data、@unpack、@file_data加载数据

import  os
from ddt import ddt,data,unpack,file_data
import unittest

def get_data():
    test_data=[{'name':'jack','age':20},{'name':'peter','age':21},{'name':'lucy','age':20},{'name':'mercy','age':20}]
    return test_data

@ddt
class MyTestCase(unittest.TestCase):
    #读取元组数据
    @data(1,2,3)
    def test_001(self,value):
        print(value)

    #读取多个数据
    @data((1,2,3),(2,4,5))
    def test_002(self,value):
        print(value)

    #读取元组数据-拆分数据
    @data((1,2,3),(2,4,5))
    @unpack
    def test_003(self,value1,value2,value3):
        print(value1,value2,value3)

    #列表
    @data([{'name':'jack','age':20},{'name':'peter','age':21},{'name':'lucy','age':20},{'name':'mercy','age':20}])
    def test_004(self,value):
        print(value)


    #字典
    @data({'name':'jack','age':20},{'name':'peter','age':21},{'name':'lucy','age':20},{'name':'mercy','age':20})
    def test_005(self,value):
        print(value)

    #字典拆分
    @data({'name':'jack','age':20},{'name':'peter','age':21},{'name':'lucy','age':20},{'name':'mercy','age':20})
    @unpack
    def test_006(self,name,age):
        print(name,age)

    #变量或者方法调用
    @data(get_data())
    def test_007(self,value):
        print(value)

    #读取文件
    @file_data(os.getcwd()+'/test.json')
    def test_008(self,value):
        print(value)


if __name__ == '__main__':
    unittest.main()
